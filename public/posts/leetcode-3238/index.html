class Solution {
public:
    int winningPlayerCount(int n, vector<vector<int>>& pick) {
        // 2.
        vector<vector<int>> arr(n,vector<int>(11,0));
        vector<bool> tags(n,false);
        int result = 0;
        for (auto pic : pick) {
            // playerId = pic[0];

            ++arr[playerId][ballColor];

        }
        return result;
    }
};
