class Solution {
private:
    const string s1 = "01010101";
    const string s2 = "10101010";
    vector<string> grid;
    void init() {
        grid.resize(8, string(8, '0'));
        for (int i = 0; i < 8; i++) {
            grid[i] = i % 2 ? s1 : s2;
        }
    }
public:
    bool checkTwoChessboards(string coordinate1, string coordinate2) {
        init();
        return grid[coordinate1[0] - 'a'][coordinate1[1] - '0' - 1] ==
               grid[coordinate2[0] - 'a'][coordinate2[1] - '0' - 1];
    }
};

class Solution {
    bool same(int a, int b) {
        if (a % 2 && b % 2)
            return true;
        else if (a % 2 == 0 && b % 2 == 0)
            return true;
        return false;
    }
public:
    bool checkTwoChessboards(string s, string t) {
        int s0 = toascii(s[0]);
        int s1 = s[1] - '0';
        int t0 = toascii(t[0]);
        int t1 = t[1] - '0';
        if (same(s0, s1) && same(t0, t1) || !same(s0, s1) && !same(t0, t1))
            return true;
        return false;
    }
};
